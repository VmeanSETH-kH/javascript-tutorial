What is the difference between Arrays and Objects?

  • Arrays use numbered indexes
    You should use arrays when you want the element names to be numbers
  • Obects use named indexes
    You should use objects when you want the element names to be strings (text).

    // The Compare Function
    function(a, b) {return a - b}

    // Numeric Sort
    const points = [40, 100, 5, 20, 99];
    points.sort(function(a,b){return a - b});

    // Sorting an Array in Random Order
    points.sort(function(){return 0.5 - Math.random()});

    // The Fisher Yates Shuffle : Sorting an Array in Random Order (2)
    for (let i = points.length -1; i > 0; i--) {
        let j = Math.floor(Math.random() * (i+1));
        let k = points[i];
        points[i] = points[j];
        points[j] = k;
    }

Array Iteration

    
    // forEach() : calls a function once for each array element
    const numbers = [40, 10, 20];
    let txt = "";
    numbers.forEach(myFunc);

    function myFunc(value, index, array) {
        txt += value + "<br>";
    }

    // map() : creates a new array by performing a function on each array element
    const num1 = [40, 10, 20];
    const num2 = num1.map(myFunc);

    function myFunc(value, index, array){
        return value * 2;
    }
    
    // flatMap() : first maps all elements of an array and then creates a new array by 
    // flattening the array
    const myArr = [1, 2, 3, 4, 5];
    const newArr = myArr.flatMap((x) => x * 2);

    // filter(): creates a new array with array elements that pass a test
    const numbers = [10, 50, 29, 66];
    const over18 = numbers.filter(myFunc);

    function myFunc(value, index, array) {
        return value > 18;
    }

    // reduce(): runs a function on each array element to produce (reduce it to) 
    // a single element
    const numbers = [10, 50, 29, 66];
    let sum = numbers.reduce(myFunc);

    function myFunc(total, value, index, array) {
        return total + value;
    } 

    // every(): check if all array values pass a test
    // some(): check if some array values pass a test
