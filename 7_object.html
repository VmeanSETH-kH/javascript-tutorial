JS Objects

Example code

    
        // Create an object
        const person = {
            firstName: "Nikola",
            lastName: "Tesla",
            id: 1122,
            fullName: function() { //fullName here is a function
                return this.firstName + " " + this.lastName;
                // <this> refers to the <person> object
            }
        }

        // Accessing Object methods
        person.fullName();  // syntax : objectName.methodName()
                            // Returns "Nikola Tesla"
