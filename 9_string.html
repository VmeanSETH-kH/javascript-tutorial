String Methods

Sample Code and Explanation

    
    // String Length
    let text = "ABCDEFGHI";
    let length = text.length;

    // Extracting String Characters
    // charAt(position) : returns the character
    let text = "HELLO WORLD";
    let char = text.charAt(0); // Returns H
    // charCodeAt(position) : return the unicode of the character
    let char = text.charCodeAt(0); //Returns 
    // at(position) : returns the character 
    const name = "Nikola Tesla";
    let letter = name.at(2); // Return k
    let character = name[2]; // Return k

    //Extracting String Parts
    // slice(start, end): extract a part of a string 
    and returns the extracted in a new string 
    let text = "Apple, Banana, Kiwi";
    let part = text.slice(7, 13); // Returns Banana
    // substring(start, end) : start and end values less than 0 
    // are treated as 0
    // substr(start, length): the second parameter specifies the 
    // length of the extracted part
    let str = "Apple, Banana, Kiwi";
    let part = str.substr(7, 6); // Returns Banana

    // Converting to Upper and Lower Case
    let text1 = "Hello World!";
    let text2 = text1.toUpperCase();
    let text3 = text1.toLowerCase();

    // String concat()
    let text1 = "Hello";
    let text2 = "World";
    let concat_text = text1.concat(" ", text2); // Returns "Hello World"
    // or
    let concat_text = text1 + " " + text2; // Recommend

    // trim(): removes whites space from both sides of a string
    let text1 = "     Hello World!    "
    let text2 = text1.trim();           // Returns "Hello World!"
    let text3 = text1.trimStart();      // Return "Hello World!    "
    let text4 = text1.trimEnd();        // Return "     Hello World!"

    // Padding
    let num = 5;
    let text = num.toString(); // convert a number to a string
    let str_length = 4;
    let padding_str = "x";
    let padded = text.padStart(str_length, padding_str); 
    // Returns "xxx5"
    let padded = text.padEnd(str_length, padding_str);
    // Returns "5xxx"

    // repeat()
    let string = "S";
    let count = 2;
    let result =  string.repeat(count); // Returns "SS"   
    
    // replace()
    let text = "Please visit Microsoft"
    let newText = text.replace("Microsoft", "Google Docs");
    // Returns "Please visit Google Docs"

    // split()
    let text = "Hi | my name is Tesla, I'm from ...,"
    let str1 = text.split(","); // Split on commas
    let str2 = text.split(" "); // Split on spaces
    let str3 = text.split("|"); // Split on pipe
